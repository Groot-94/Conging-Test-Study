func solution(_ t:String, _ p:String) -> Int {
    var resultT = ""
    var result = 0
    var arrayT = t.map {
        String($0)
    }
    
    for i in 0...arrayT.count - p.count{
        for j in 0...p.count-1{
            resultT += arrayT[i+j]
        }
        
        if Int(resultT)! <= Int(p)!{
            result += 1
        }
        resultT = ""
    }
    
    return result
}

print(solution("3141592", "271"))
